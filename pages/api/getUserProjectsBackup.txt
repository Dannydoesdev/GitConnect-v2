// pages/api/getUserProjects.js
import { getAllUserProjectsWithUsernameLowercase } from 'path/to/your/firebase/functions';
import { getProfileDataWithUsernameLowercase } from 'path/to/your/firebase/functions';


export default async (req, res) => {
  const { username } = req.query;
  try {
    const data = await getAllUserProjectsWithUsernameLowercase(username);
    res.status(200).json(data);
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
};
